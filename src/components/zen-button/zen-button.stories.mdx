import { Meta, Story, Canvas, ArgsTable } from '@storybook/addon-docs/blocks';
import { html } from 'lit-html';
import Readme from './readme.md';
import { getArgTypes, getDefaultArgs, getComponentData, spreadArgs } from '../../../.storybook/helpers/argTypes';

import data from '../../../stencilDocs.json';
const compData = data.components.find(n => n.tag === 'zen-button');
const argTypes = getArgTypes(compData);

import { litHtmlIcon, styles } from '../helpers/fa-icons';
import { faUser, faChevronDown } from '@fortawesome/pro-light-svg-icons';

<Meta title="Buttons/Button" component="zen-button" argTypes={argTypes} />

export const StoryWithControls = args => html/*html*/ ` <zen-button ...="${spreadArgs(args)}"></zen-button>`;

export const StorySimple = () =>
  html/*html*/ `
    <div style="display: flex; justify-content: space-evenly;">
      <zen-button variant="primary"></zen-button><zen-button variant="secondary"></zen-button
      ><zen-button variant="tertiary"></zen-button><zen-button variant="positive"></zen-button
      ><zen-button variant="destructive"></zen-button>
    </div>
  `;

export const StoryLoading = () => html/*html*/ ` <zen-button loading></zen-button> `;

export const StoryDisabled = () => html/*html*/ ` <zen-button disabled></zen-button> `;

export const StoryWithLeadingSlot = () =>
  html/*html*/ `
    <zen-button>
      <i slot="leadingIcon" style="width: 0.625rem;" .innerHTML="${litHtmlIcon(faUser)}"></i>
    </zen-button>
  `;

export const StoryWithTrailingSlot = () =>
  html/*html*/ `
    <zen-button
      ><i slot="trailingIcon" style="width: 0.625rem;" .innerHTML="${litHtmlIcon(faChevronDown)}"></i
    ></zen-button>
  `;

export const StoryWithLeadingAndTrailingSlots = () =>
  html/*html*/ `
    <zen-button>
      <i slot="leadingIcon" style="width: 0.625rem;" .innerHTML="${litHtmlIcon(faUser)}"></i>
      Leading and Trailing Slots
      <i slot="trailingIcon" style="width: 0.625rem;" .innerHTML="${litHtmlIcon(faChevronDown)}"></i>
    </zen-button>
  `;
