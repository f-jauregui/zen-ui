export const StoryMultipleModals = () => {
  function showEditor() {
    // close first to enable infinite loop:
    document.querySelector('#editor').show = false;
    document.querySelector('#editor').show = true;
  }
  function showConfirm() {
    document.querySelector('#confirm').show = false;
    document.querySelector('#confirm').show = true;
  }
  function closeEditor() {
    document.querySelector('#editor').show = false;
  }
  return html/*html*/ `
    <div style="display: flex; flex-direction: row;">
      <sb-zen-button @click="${showEditor}">Show Editor</sb-zen-button>
      <sb-zen-button @click="${showConfirm}" class="ml-4">Show Confirmation</sb-zen-button>
    </div>
    <sb-zen-modal id="editor" label="Risk score editor" @cancel="${closeEditor}">
      <sb-zen-text>This is a modal window editor example.</sb-zen-text>
      <sb-zen-space spacing="sm" vertical>
        <sb-zen-text variant="label" required>Username</sb-zen-text>
        <sb-zen-input></sb-zen-input>
        <sb-zen-text variant="support"> This field should be unique </sb-zen-text>
      </sb-zen-space>
      <div slot="buttons">
        <sb-zen-button @click="${closeEditor}" variant="secondary">Cancel</sb-zen-button>
        <sb-zen-button @click="${showConfirm}" class="ml-4" variant="secondary">Delete</sb-zen-button>
        <sb-zen-button @click="${closeEditor}" class="ml-4">Accept</sb-zen-button>
      </div>
    </sb-zen-modal>
    <sb-zen-modal
      id="confirm"
      label="Confirmation"
      @cancel="${e => {
        e.target.show = false;
      }}"
      @ok="${e => {
        e.target.show = false;
      }}"
    >
      <p>Are you sure?!</p>
      <sb-zen-button @click="${showEditor}" variant="secondary">Show Editor</sb-zen-button>
    </sb-zen-modal>
  `;
};
