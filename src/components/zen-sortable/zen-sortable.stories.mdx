import { html } from 'lit-html';
import { Meta, Description, Story, Canvas, ArgsTable } from '@storybook/addon-docs/blocks';
import { waitForElement } from '../../../.storybook/helpers/utils';
import { getArgTypesAndArgs, getDefaultArgs, getComponentData, spreadArgs } from '../../../.storybook/helpers/argTypes';
const { argTypes } = getArgTypesAndArgs('zen-sortable');

<Meta title="In progress/Sortable" component="zen-sortable" argTypes={argTypes} />

export const data = [
  {
    name: 'Title',
    selected: true,
  },
  {
    name: 'Description',
    selected: true,
  },
  {
    name: 'Status',
    selected: true,
  },
  {
    name: 'Manager',
    selected: false,
  },
];

export const StorySortable = () => {
  waitForElement('#sortable', element => {
    element.data = data;
  });
  return html/*html*/ `<sb-zen-sortable
    style="width:320px;border: 1px solid lightgray;"
    id="sortable"
  ></sb-zen-sortable>`;
};

export const StoryWithControls = args => {
  waitForElement('#control', element => {
    element.data = data;
  });
  return html/*html*/ `<sb-zen-sortable
    style="border: 1px solid lightgray;"
    padddig="md"
    id="control"
    ...="${spreadArgs(args, argTypes)}"
  ></sb-zen-sortable>`;
};
